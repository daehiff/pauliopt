 25class Coupling(FrozenSet[int]):
 26    """
 27    Type for couplings in a qubit topology, i.e. unordered
 28    pairs of adjacent qubits.
 29    """
 30
 31    def __new__(cls, fst: int, snd: int) -> "Coupling":
 32        if not isinstance(fst, int):
 33            raise TypeError(f"Expected integer, found {fst}")
 34        if not isinstance(snd, int):
 35            raise TypeError(f"Expected integer, found {snd}")
 36        if len({fst, snd}) != 2:
 37            raise ValueError("Expected a pair of distinct qubits.")
 38        # see https://github.com/python/mypy/issues/6061
 39        return super(Coupling, cls).__new__(cls, {fst, snd})  # type: ignore
 40
 41    @property
 42    def as_pair(self) -> Tuple[int, int]:
 43        """
 44        Returns the coupling as a (increasingly) ordered pair.
 45        """
 46        return (min(*self), max(*self))  # pylint: disable = not-an-iterable
 47
 48    def __str__(self) -> str:
 49        fst, snd = sorted(self)
 50        return f"{{{fst}, {snd}}}"
 51
 52    def __repr__(self) -> str:
 53        fst, snd = sorted(self)
 54        return f"Coupling({fst}, {snd})"

 58"""
 59    Type alias for things that could be used to specify couplings,
 60    namely any collection of `int` (subject to additional restrictions).
 61
 62    In an ideal world, this should be "int collections of len 2",
 63    but static typing does not yet allow for such a constraint.

 67def _validate_coupling(num_qubits: int, coupling_like: CouplingLike) -> Coupling:
 68    # Assumes `qubits` was already validated.
 69    if not isinstance(coupling_like, Collection) or len(coupling_like) != 2:
 70        raise TypeError(f"Expected a pair, found {coupling_like}.")
 71    fst, snd = coupling_like
 72    if not 0 <= fst < num_qubits:
 73        raise TypeError(f"Invalid qubit {fst}.")
 74    if not 0 <= snd < num_qubits:
 75        raise TypeError(f"Invalid qubit {snd}.")
 76    return Coupling(fst, snd)

110def _floyd_warshall(topology: "Topology"):
111    next = np.ones((topology.num_qubits, topology.num_qubits), dtype=int)
112    dist = np.inf * np.ones((topology.num_qubits, topology.num_qubits))
113    G = topology.to_nx
114    for u, v in G.edges():
115        dist[u, v] = 1.0
116        dist[v, u] = 1.0
117        next[u, v] = v
118        next[v, u] = u
119    for v in G.nodes:
120        dist[v, v] = 0
121        next[v, v] = v
122    for k in G.nodes:
123        for i in G.nodes:
124            for j in G.nodes:
125                if dist[i][j] > dist[i][k] + dist[k][j]:
126                    dist[i][j] = dist[i][k] + dist[k][j]
127                    next[i][j] = next[i][k]
128    return dist, next

131class Topology:
132    """
133    Container class for a qubit topology.
134    """
135
136    _num_qubits: int
137    _couplings: FrozenSet[Coupling]
138    _adjacent: Tuple[FrozenSet[int], ...]
139    _dist: np.ndarray
140    _named: Optional[str] = None
141
142    def __init__(self, num_qubits: int, couplings: Collection[CouplingLike]):
143        if not isinstance(num_qubits, int) or num_qubits <= 0:
144            raise TypeError("Number of qubits must be a positive integer.")
145        if not isinstance(couplings, Collection):
146            raise TypeError("Couplings must be a collection of pairs of qubits.")
147        self._num_qubits = num_qubits
148        self._couplings = frozenset(
149            _validate_coupling(num_qubits, c) for c in couplings
150        )
151        _adjacent: Tuple[Set[int], ...] = tuple(set() for _ in range(num_qubits))
152        for fst, snd in self.couplings:
153            _adjacent[fst].add(snd)
154            _adjacent[snd].add(fst)
155        self._adjacent = tuple(frozenset(n) for n in _adjacent)
156        self._dist, self._next = _floyd_warshall(self)

193    def is_planar(self) -> bool:
194        """
195        Whether this qubit topology is a planar graph.
196        """
197        try:
198            # pylint: disable = import-outside-toplevel

202        G = self.to_nx

216    def to_nx(self):
217        """
218        Readonly property returning a NetworkX graph version of this topology.
219        Requires the 'networkx' library to work.
220        """
221        try:
222            # pylint: disable = import-outside-toplevel
223            import networkx as nx  # type: ignore

226        g = nx.Graph()
227        g.add_nodes_from(sorted(self.qubits))
228        g.add_edges_from(sorted(self.couplings, key=lambda c: c.as_pair))

231    def draw(
232        self,
233        layout: str = "kamada_kawai",
234        *,
235        figsize: Optional[Tuple[int, int]] = None,
236        filename: Optional[str] = None,
237        **kwargs,
238    ):
239        """
240        Draws this qubit topology using NetworkX and Matplotlib.
241
242        The `layout` keyword argument can be used to select a NetworkX layout
243        from the available ones (exposed by `Topology.available_nx_layouts`).
244        The `figsize` keyword argument is passed to `matplotlib.pyplot.figure`:
245        if specified, it determines the width and height of the figure being drawn.
246        Keyword arguments `kwargs` are those of `networkx.draw_networkx`.
247        If the keyword argument `filename` is set, the figure is also saved.
248        """
249        try:
250            # pylint: disable = import-outside-toplevel
251            import networkx as nx  # type: ignore
252        except ModuleNotFoundError as _:
253            raise ModuleNotFoundError("You must install the 'networkx' library.")
254        try:
255            # pylint: disable = import-outside-toplevel
256            import matplotlib.pyplot as plt  # type: ignore
257        except ModuleNotFoundError as _:
258            raise ModuleNotFoundError("You must install the 'matplotlib' library.")
259        G = self.to_nx
260        kwargs = {**kwargs}
261        layouts = self.available_nx_layouts
262        if "pos" not in kwargs:
263            if layout not in layouts:
264                raise ValueError(
265                    f"Invalid layout found: {layout}. "
266                    f"Valid layouts: {', '.join(repr(l) for l in layouts)}"
267                )
268            kwargs["pos"] = getattr(nx, layout + "_layout")(G)
269        if "node_color" not in kwargs:
270            kwargs["node_color"] = "#dddddd"
271        plt.figure(figsize=figsize)
272        nx.draw_networkx(G, **kwargs)
273        if filename is not None:
274            plt.savefig(filename)
275        plt.show()

277    def adjacent(self, qubit: int) -> FrozenSet[int]:
278        """
279        Readonly property exposing the (frozen) set of qubits adjacent
280        to (i.e. couple with) the given qubit.
281        """
282        if not isinstance(qubit, int):
283            raise TypeError("Qubit should be an integer.")
284        if qubit not in self:
285            raise ValueError(f"Invalid qubit {qubit}.")
286        return self._adjacent[qubit]

288    def incident(self, qubit: int) -> Iterator[Coupling]:
289        """
290        Readonly property returning an iterator running over all couplings
291        incident onto the given qubit.
292
293        This is returned as an iterator, rather than a collection,
294        because the couplings are generated on the fly (i.e. this is not
295        merely exposing some internal collection).
296        """
297        adjacent = self.adjacent(qubit)
298        return (Coupling(qubit, q) for q in adjacent)

300    def dist(self, fro: int, to: int) -> int:
301        """
302        Returns the distance between two given qubits in the topology.
303        """
304        if not isinstance(fro, int) or fro not in self:
305            raise TypeError(f"Expected a valid qubit, found {fro}.")
306        if not isinstance(to, int) or to not in self:
307            raise TypeError(f"Expected a valid qubit, found {to}.")
308        return self._dist[fro, to]

310    def mapped_fwd(self, mapping: Union[Sequence[int], Dict[int, int]]) -> "Topology":
311        """
312        Returns a topology with the same couplings, but remapping the qubits using
313        the given mapping.
314        """
315        if isinstance(mapping, Sequence):
316            if len(mapping) < self.num_qubits:
317                raise ValueError(
318                    f"Expected mapping keys [0,...,{self._num_qubits}], "
319                    f"found {sorted(mapping)} instead."
320                )
321            _mapping = list(mapping)
322        elif isinstance(mapping, Mapping):
323            _mapping = []
324            for i in range(self._num_qubits):
325                if i not in mapping:
326                    raise ValueError(
327                        f"Expected mapping keys [0,...,{self._num_qubits}], "
328                        f"found {sorted(mapping.keys())} instead."
329                    )
330                _mapping.append(mapping[i])
331        else:
332            raise TypeError(
333                f"Expected Sequence[int] or Mapping[int, int], "
334                f"found {type(mapping)} instead."
335            )
336        if set(_mapping) != set(range(self._num_qubits)):
337            raise ValueError(
338                f"Expected mapping values [0,...,{self._num_qubits}], "
339                f"found {sorted(_mapping)} instead."
340            )
341        mapped_couplings = [
342            {_mapping[x] for x in coupling} for coupling in self._couplings
343        ]
344        return Topology(self.num_qubits, mapped_couplings)
345
346    def shortest_path(self, fro: int, to: int):
347        """
348        Computes the shortest path using the next lookup table from the Floyd–Warshall algorithm
349        """
350        if self._next[fro, to] is None:
351            raise Exception("Unconnected Architecture")
352        else:
353            path = [fro]
354            while fro != to:
355                fro = self._next[fro, to]
356                path.append(fro)
357            return path

359    def steiner_tree(self, terminals: list[int], subgraph: list[int] = []):
360        """
361        Computes the Steiner tree over the topology with given terminals.
362        `subgraph` can be used to find the Steiner tree of a subgraph of the topology.
363        Requires networkx to be installed to work.
364        Returns a networkx Graph.
365        """
366        try:
367            # pylint: disable = import-outside-toplevel
368            from networkx.algorithms.approximation.steinertree import steiner_tree
369        except ModuleNotFoundError as _:
370            raise ModuleNotFoundError("You must install the 'networkx' library.")
371        if subgraph:
372            G = self.to_nx.subgraph(subgraph)
373        else:
374            G = self.to_nx
375        return steiner_tree(G, terminals)

377    def non_cutting_qubits(self, subgraph: List[int] = []) -> List[int]:
378        try:
379            # pylint: disable = import-outside-toplevel
380            from networkx import articulation_points
381        except ModuleNotFoundError as _:
382            raise ModuleNotFoundError("You must install the 'networkx' library.")
383        if subgraph:
384            G = self.to_nx.subgraph(subgraph)
385        else:
386            G = self.to_nx
387            subgraph = self.qubits
388        cut_vertices = list(articulation_points(G))
389        return [q for q in subgraph if q not in cut_vertices]

391    def mapped_bwd(self, mapping: Union[Sequence[int], Dict[int, int]]) -> "Topology":
392        """
393        Returns a topology with the same couplings, but remapping the qubits using
394        the inverse of the given mapping.
395        """
396        if isinstance(mapping, Sequence):
397            if len(mapping) < self.num_qubits:
398                raise ValueError(
399                    f"Expected mapping keys [0,...,{self._num_qubits}], "
400                    f"found {sorted(mapping)} instead."
401                )
402            _rev_mapping = {mapping[i]: i for i in mapping}
403        elif isinstance(mapping, Mapping):
404            _rev_mapping = {}
405            for i in range(self._num_qubits):
406                if i not in mapping:
407                    raise ValueError(
408                        f"Expected mapping keys [0,...,{self._num_qubits}], "
409                        f"found {sorted(mapping.keys())} instead."
410                    )
411                _rev_mapping[mapping[i]] = i
412        else:
413            raise TypeError(
414                f"Expected Sequence[int] or Mapping[int, int], "
415                f"found {type(mapping)} instead."
416            )
417        if set(_rev_mapping.keys()) != set(range(self._num_qubits)):
418            raise ValueError(
419                f"Expected mapping values [0,...,{self._num_qubits}], "
420                f"found {sorted(_rev_mapping.keys())} instead."
421            )
422        mapped_couplings = [
423            {_rev_mapping[x] for x in coupling} for coupling in self._couplings
424        ]
425        return Topology(self.num_qubits, mapped_couplings)

427    def __contains__(self, x: Union[int, Coupling, Tuple[int, int]]) -> bool:
428        if isinstance(x, int):
429            return 0 <= x < self._num_qubits
430        if isinstance(x, Coupling):
431            return x in self._couplings
432        if isinstance(x, tuple) and len(x) == 2:
433            fst, snd = x
434            if isinstance(fst, int) and isinstance(snd, int):
435                return Coupling(fst, snd) in self
436        raise TypeError(f"Expected qubit or coupling, found {x}")

460    @staticmethod
461    def from_dict(topology: Union[TopologyDict, str]) -> "Topology":
462        """
463        Creates a `Topology` instance from a dictionary in the
464        format obtained from `Topology.as_dict`,
465        for de-serialization purposes.
466        """
467        if isinstance(topology, str):
468            line_pattern = re.compile(r"line\(([0-9]+)\)")
469            if match := line_pattern.match(topology):
470                return Topology.line(int(match[1]))
471            cycle_pattern = re.compile(r"cycle\(([0-9]+)\)")
472            if match := cycle_pattern.match(topology):
473                return Topology.cycle(int(match[1]))
474            complete_pattern = re.compile(r"complete\(([0-9]+)\)")
475            if match := complete_pattern.match(topology):
476                return Topology.complete(int(match[1]))
477            grid_pattern = re.compile(r"grid\(([0-9]+),([0-9]+)\)")
478            if match := grid_pattern.match(topology):
479                return Topology.grid(int(match[1]), int(match[2]))
480            periodic_grid_pattern = re.compile(r"periodic_grid\(([0-9]+),([0-9]+)\)")
481            if match := periodic_grid_pattern.match(topology):
482                return Topology.periodic_grid(int(match[1]), int(match[2]))
483            raise ValueError(f"Unexpected special topology {topology}")
484        if "num_qubits" not in topology:
485            raise TypeError("Expected key 'qubits'.")
486        if "couplings" not in topology:
487            raise TypeError("Expected key 'couplings'.")
488        return Topology(topology["num_qubits"], topology["couplings"])

528    @staticmethod
529    def grid(num_rows: int, num_cols: int) -> "Topology":
530        """
531        Creates a grid topology with the given number of rows and cols.
532        Qubits are indexed by rows.
533        """
534        if not isinstance(num_rows, int) or num_rows <= 0:
535            raise TypeError("Number of rows must be positive integer.")
536        if not isinstance(num_cols, int) or num_cols <= 0:
537            raise TypeError("Number of cols must be positive integer.")
538        num_qubits = num_rows * num_cols
539
540        def qubit(r, c):
541            return num_cols * r + c
542
543        couplings: List[List[int]] = []
544        for r in range(num_rows):
545            for c in range(num_cols):
546                if r < num_rows - 1:
547                    couplings.append([qubit(r, c), qubit(r + 1, c)])
548                if c < num_cols - 1:
549                    couplings.append([qubit(r, c), qubit(r, c + 1)])
550        top = Topology(num_qubits, couplings)
551        top._named = (
552            f"grid({num_rows},{num_cols})"  # pylint: disable = protected-access
553        )
554        return top

728    def _flip(self, coupling: Coupling) -> "Matching":

732                f"one of the qubits is already matched. "
733            )
734        fst, snd = coupling

736            self._matched_couplings.remove(coupling)

738            self._matched_qubits.remove(snd)
739            self._incident_coupling[fst] = None

741        else:
742            self._matched_couplings.add(coupling)
743            self._matched_qubits.add(fst)

745            self._incident_coupling[fst] = coupling
746            self._incident_coupling[snd] = coupling
747        return self

753                yield coupling
754            elif fst not in self._matched_qubits and snd not in self._matched_qubits:
