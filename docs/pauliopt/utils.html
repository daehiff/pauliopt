 30def calculate_orthogonal_point(a, b, d, left):
 31    direction_vector = b - a
 32    magnitude = np.linalg.norm(direction_vector)
 33    normalized_direction_vector = direction_vector / magnitude
 34    if left:
 35        orthogonal_vector = np.array(
 36            [-normalized_direction_vector[1], normalized_direction_vector[0]]
 37        )
 38    else:
 39        orthogonal_vector = np.array(
 40            [normalized_direction_vector[1], -normalized_direction_vector[0]]
 41        )
 42    midpoint = (a + b) / 2
 43    orthogonal_point = midpoint + d * orthogonal_vector
 44    return int(orthogonal_point[0]), int(orthogonal_point[1])

356    @staticmethod
357    def random(
358        subdivision: int = 4,
359        *,
360        size: int = 1,
361        rng_seed: Optional[int] = None,
362        nonzero: bool = False,
363    ) -> Union["Angle", Tuple["Angle", ...]]:
364        """
365        Generates a random angle with the given `subdivision`:
366        `r * pi/subdivision` for random `r in range(2*subdivision)`.
367        """
368        if not isinstance(subdivision, int):
369            raise TypeError()
370        if not isinstance(size, int):
371            raise TypeError()
372        if not isinstance(nonzero, bool):
373            raise TypeError()
374        if subdivision <= 0:
375            raise ValueError("Subdivision must be positive.")
376        if size <= 0:
377            raise ValueError("Size must be positive.")
378        if rng_seed is not None and not isinstance(rng_seed, int):
379            raise TypeError("RNG seed must be integer or 'None'.")
380        rng = np.random.default_rng(seed=rng_seed)
381        if nonzero:
382            rs = 1 + rng.integers(
383                2 * subdivision - 1, size=size
384            )  # type: ignore[attr-defined]
385        else:
386            rs = rng.integers(2 * subdivision, size=size)  # type: ignore[attr-defined]
387        if size == 1:
388            return Angle(Fraction(int(rs[0]), subdivision))
389        return tuple(Angle(Fraction(int(r), subdivision)) for r in rs)

409def SumprodAngleExpr(
410    *exprs: AngleExpr, coeffs: Union[int, Fraction, Sequence[Union[int, Fraction]]] = 1
411) -> AngleExpr:
412    if not isinstance(coeffs, Sequence):
413        coeffs = (coeffs,)
414    if len(coeffs) != len(exprs):
415        raise ValueError(
416            f"Expected a sequence of {len(exprs)} coefficients, "
417            f"found {len(coeffs)}."
418        )
419    _coeffs: Dict[AngleExpr, Fraction] = {}
420    _const: Angle = Angle.zero
421    for e, c in zip(exprs, coeffs):
422        if isinstance(e, Angle):
423            _const += e._mul(c)  # pylint: disable = protected-access
424        elif isinstance(e, _SumprodAngleExpr):
425            for sub_e, sub_c in e.coeffs.items():
426                new_c = c * sub_c
427                if sub_e in _coeffs:
428                    new_c += _coeffs[sub_e]
429                _coeffs[sub_e] = new_c
430        elif c != 0:
431            c = Fraction(c)
432            if e in _coeffs:
433                c += _coeffs[e]
434            _coeffs[e] = c
435    _coeffs = {
436        e: c
437        for e, c in sorted(_coeffs.items(), key=lambda i: -i[1])
438        if c != 0 and not e.is_zero
439    }
440    if not _coeffs:
441        return _const
442    return _SumprodAngleExpr(_coeffs, _const)

496            )
