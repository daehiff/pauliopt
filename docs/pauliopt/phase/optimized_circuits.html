 76def _validate_temp_schedule(
 77    schedule: Union[StandardTempSchedule, TempSchedule]
 78) -> TempSchedule:
 79    if not isinstance(schedule, TempSchedule):
 80        if not isinstance(schedule, tuple) or len(schedule) != 3:
 81            raise TypeError(
 82                f"Expected triple (schedule_name, t_init, t_final), "
 83                f"found {schedule}"
 84            )
 85        schedule_name, t_init, t_final = schedule
 86        if schedule_name not in StandardTempSchedules:
 87            raise TypeError(
 88                f"Invalid standard temperature schedule name {schedule_name}, "
 89                f"allowed names are: {list(StandardTempSchedules.keys())}"
 90            )
 91        if not isinstance(t_init, (int, float)) or not isinstance(
 92            t_final, (int, float)
 93        ):
 94            raise TypeError("Expected t_init and t_final to be int or float.")
 95        schedule = StandardTempSchedules[schedule_name](t_init, t_final)
 96    return schedule

118class OptimizedPhaseCircuit:
119    # pylint: disable = too-many-instance-attributes
120    """
121    Container for a phase circuit to be progressively optimized.
122    The original phase circuit is passed to the constructor, together
123    with a qubit topology and a fixed number of layers constraining the
124    CX circuit to be used for simplification.
125
126    To understand the structure of the optimized phase circuit,
127    consider the following code snippet:
128
129    ```py
130        opt_circ = PhaseCircuitCXBlockOptimizer(orig_circ, topology, num_cx_layers)
131        # perform optimization using the methods of `opt_circ`
132        phase_block = opt_circ.phase_block
133        cx_block = opt_circ.cx_block
134    ```
135
136    The optimized circuit is obtained by composing three blocks:
137
138    1. a first block of CX gates, given by `cx_block.dag()`
139       (the same CX gates of `cx_block`, but in reverse order);
140    2. a central block of phase gadgets, given by `phase_block`;
141    3. a final block of CX gates, given by `cx_block`.
142
143    An optional keyword argument `circuit_rep` (default: 1) can be passed to the
144    constructor to indicate that the original circuit is to be repeated a certain
145    number of times (default: 1). In the optimized circuit, this is achieved by
146    repeating the `phase_block` part (at point 2. above) a number of times
147    given by the `circuit_rep` argument.
148    The first and last CX blocks are left unaltered (because the intermediate CX
149    blocks would cancel each other out in pairs when repeating the optimized circuit).
150    """
151

153    _circuit_rep: int
154    _phase_block: PhaseCircuit

177        *,

327    def to_qiskit(self, simplified: bool = True):
328        """
329        Returns the optimized circuit as a Qiskit circuit.
330
331        This method relies on the `qiskit` library being available.
332        Specifically, the `circuit` argument must be of type
333        `qiskit.providers.BaseBackend`.
334        """
335        try:
336            # pylint: disable = import-outside-toplevel
337            from qiskit.circuit import QuantumCircuit  # type: ignore
338        except ModuleNotFoundError as _:
339            raise ModuleNotFoundError("You must install the 'qiskit' library.")
340        if simplified:
341            self.simplify()
342        circuit = QuantumCircuit(self.num_qubits)
343        front_cx_block = CXCircuit(self.topology, self._cx_block)
344        synthesized_cx_block1 = front_cx_block.to_qiskit(
345            self._cx_method, reallocate=self._reallocate
346        )
347        synthesized_phase_block, cxs = self._phase_block.to_qiskit(
348            self.topology,
349            simplified=simplified,
350            method=self._phase_method,
351            return_cx=True,
352        )
353        for layer in self._cx_block:
354            cxs >>= layer
355        synthesized_cx_block2 = cxs.to_qiskit(
356            self._cx_method, reallocate=self._reallocate
357        )
358        circuit.compose(synthesized_cx_block1.inverse(), inplace=True)
359        for _ in range(self._circuit_rep):
360            circuit.compose(synthesized_phase_block, inplace=True)
361        circuit.compose(synthesized_cx_block2, inplace=True)
362        if self._reallocate:
363            undo_mapping = [
364                self._qubit_mapping.index(i) for i in range(self.num_qubits)
365            ]
366            initial_layout = [
367                undo_mapping[i] for i in synthesized_cx_block1.metadata["final_layout"]
368            ]
369            final_layout = [
370                undo_mapping[i] for i in synthesized_cx_block2.metadata["final_layout"]
371            ]
372            circuit.metadata = {
373                "initial_layout": initial_layout,
374                "final_layout": final_layout,
375            }
376        return circuit

378    def simplify(self):
379        """
380        Simplifies the phase block according to the commutation and fusion
381        rules for phase gadgets.
382        """
383        new_phase_block = self._phase_block.cloned()
384        for _ in range(1, self._circuit_rep):
385            new_phase_block >>= self._phase_block
386        self._phase_block = new_phase_block.simplified()
387        self._phase_block_view = PhaseCircuitView(self._phase_block)
388        self._circuit_rep = 1
389        self._gadget_cx_count_cache = {}
390        self._cx_count, self._cx_blocks_count = self._compute_cx_count()

392    def anneal(
393        self,
394        num_iters: int,
395        *,
396        schedule: Union[StandardTempSchedule, TempSchedule] = ("linear", 1.0, 0.1),
397        loggers: AnnealingLoggers = {},
398    ):
399        # pylint: disable = dangerous-default-value
400        # pylint: disable = too-many-locals
401        """
402        Performs a cycle of simulated annealing optimization,
403        using the given number of iterations, temperature schedule,
404        initial/final temperatures.
405
406        The circuit is modified in-place and then returned, as per the
407        [fluent API pattern](https://en.wikipedia.org/wiki/Fluent_interface).
408        """
409        # Validate arguments:
410        if not isinstance(num_iters, int) or num_iters <= 0:
411            raise TypeError(f"Expected a positive integer, found {num_iters}.")
412        schedule = _validate_temp_schedule(schedule)
413        log_start, log_iter, log_end = _validate_loggers(loggers)
414        # Log start:
415        if log_start is not None:
416            log_start(self._cx_count, num_iters)
417        # Pre-sample random numbers to use in iterations:
418        rand = self._rng.uniform(size=num_iters)
419        # Run iterations:
420        for it in range(num_iters):
421            old_output_mapping = [i for i in self._output_mapping]
422            t = schedule(it, num_iters=num_iters)
423            layer_idx, (ctrl, trgt) = self.random_flip_cx()
424            new_cx_count, new_cx_blocks_count = self._compute_cx_count()
425            cx_count_diff = new_cx_count - self._cx_count
426            accept_step = cx_count_diff < 0 or rand[it] < np.exp(
427                -np.log(2) * cx_count_diff / t
428            )
429            if log_iter is not None:
430                log_iter(
431                    it,
432                    self._cx_count,
433                    new_cx_count,
434                    accept_step,
435                    (layer_idx, (ctrl, trgt)),
436                    t,
437                    num_iters,
438                )
439            if accept_step:
440                # Accept changes:
441                self._cx_count = new_cx_count
442                self._cx_blocks_count = new_cx_blocks_count
443            else:
444                # Undo changes:
445                self._flip_cx(layer_idx, ctrl, trgt)
446                if self._reallocate:
447                    self._output_mapping = old_output_mapping
448        # Log end:
449        if log_end is not None:
450            log_end(self._cx_count, num_iters)

452    def random_flip_cx(self) -> Tuple[int, Tuple[int, int]]:
453        """
454        Randomly flips a CX gate in the CX circuit used for the optimization,
455        updating both the CX circuit and the circuit being optimized.
456
457        Returns the layer index and gate (pair of control and target) that were
458        flipped (e.g. in case the flip needs to be subsequently undone).
459        """
460        while True:
461            layer_idx = int(self._rng.integers(len(self._cx_block)))
462            ctrl, trgt = self._cx_block[layer_idx].random_flip_cx(self._rng)
463            if layer_idx < len(self._cx_block) - 1 and self._cx_block[
464                layer_idx + 1
465            ].has_cx(ctrl, trgt):
466                # Try again if CX gate already present in layer above (to avoid redundancy)
467                continue
468            if layer_idx > 0 and self._cx_block[layer_idx - 1].has_cx(ctrl, trgt):
469                # Try again if CX gate already present in layer below (to avoid redundancy)
470                continue
471            self._flip_cx(layer_idx, ctrl, trgt)
472            return layer_idx, (ctrl, trgt)

474    def is_cx_flippable(self, layer_idx: int, ctrl: int, trgt: int) -> bool:
475        """
476        Checks whether the given CX gate can be flipped in the given layer.
477        """
478        if not isinstance(layer_idx, int) or not 0 <= layer_idx < len(self._cx_block):
479            raise TypeError(f"Invalid layer index {layer_idx} for CX circuit.")
480        layer = self._cx_block[layer_idx]
481        return layer.is_cx_flippable(ctrl, trgt)

483    def flip_cx(self, layer_idx: int, ctrl: int, trgt: int):
484        """
485        Performs the actions needed to flip the given CX gate in the given layer
486        of the CX circuit used for the optimization:
487
488        - undoes all gates in layers subsequent to the given layer which are
489          causally following the given gate, starting from the last layer and
490          working backwards towards the gate;
491        - applies the desired gate;
492        - redoes all gate undone, in reverse order (starting from the gate and
493          working forwards towards the last layer).
494        """
495        if not self.is_cx_flippable(layer_idx, ctrl, trgt):
496            raise ValueError(
497                f"Gate {(ctrl, trgt)} cannot be flipped in layer number {layer_idx}."
498            )
499        self._flip_cx(layer_idx, ctrl, trgt)

501    def _flip_cx(self, layer_idx: int, ctrl: int, trgt: int) -> None:
502        conj_by: Deque[Tuple[int, int]] = deque([(ctrl, trgt)])
503        qubits_spanned: Set[int] = set([ctrl, trgt])
504        for layer in reversed(self._cx_block[:layer_idx]):
505            new_qubits_spanned: Set[int] = set()
506            incident_gates: Set[Tuple[int, int]] = set()
507            for q in qubits_spanned:
508                incident_gate = layer.incident(q)
509                if incident_gate is not None:
510                    incident_gates.add(incident_gate)
511                    new_qubits_spanned.update({incident_gate[0], incident_gate[1]})
512            for incident_gate in incident_gates:
513                conj_by.appendleft(incident_gate)  # will first undo the gate ...
514                # ... then do all gates already in conj_by ...
515                conj_by.append(incident_gate)  # ... then finally redo the gate
516            qubits_spanned.update(new_qubits_spanned)
517        # Flip the gate in the CX circuit:
518        self._cx_block[layer_idx].flip_cx(ctrl, trgt)
519        # Conjugate the optimized phase gadget circuit by all necessary gates:
520        for cx in conj_by:
521            self._phase_block.conj_by_cx(*cx)

523    def _compute_cx_count(self) -> Tuple[int, int]:
524        # pylint: disable = protected-access
525        self._output_mapping = list(range(self.num_qubits))
526        if self._cx_method == "naive":
527            front_cx_count = self._cx_block.num_gates
528        else:
529            front_cx_count = CXCircuit.from_parity_matrix(
530                self._cx_block.parity_matrix(),
531                self.topology,
532                reallocate=self._reallocate,
533                method=self._cx_method,
534            ).num_gates
535        phase_qiskit, cxs = self._phase_block.to_qiskit(
536            self.topology,
537            simplified=False,
538            method=self._phase_method,
539            cx_synth="naive",
540            reallocate=self._reallocate,
541            return_cx=True,
542        )
543        counts = phase_qiskit.count_ops()
544        if "cx" in counts:
545            phase_cx_count = counts["cx"]
546        else:
547            phase_cx_count = 0
548        cxs >>= self._cx_block
549        if self._cx_method == "naive":
550            new_cxs = cxs
551        else:
552            new_cxs = CXCircuit.from_parity_matrix(
553                cxs.parity_matrix(),
554                self.topology,
555                reallocate=self._reallocate,
556                method=self._cx_method,
557            )
558        back_cx_count = new_cxs.num_gates
559        cx_count = front_cx_count + phase_cx_count + back_cx_count
560        if self._reallocate:
561            self._output_mapping = new_cxs._output_mapping
562        return cx_count, 2 * front_cx_count

122    Container for a phase circuit to be progressively optimized.
123    The original phase circuit is passed to the constructor, together
124    with a qubit topology and a fixed number of layers constraining the
125    CX circuit to be used for simplification.
126
127    To understand the structure of the optimized phase circuit,
128    consider the following code snippet:
129
130    ```py
131        opt_circ = PhaseCircuitCXBlockOptimizer(orig_circ, topology, num_cx_layers)
132        # perform optimization using the methods of `opt_circ`
133        phase_block = opt_circ.phase_block
134        cx_block = opt_circ.cx_block
135    ```
136
137    The optimized circuit is obtained by composing three blocks:
138
139    1. a first block of CX gates, given by `cx_block.dag()`
140       (the same CX gates of `cx_block`, but in reverse order);
141    2. a central block of phase gadgets, given by `phase_block`;
142    3. a final block of CX gates, given by `cx_block`.
143

145    constructor to indicate that the original circuit is to be repeated a certain

148    given by the `circuit_rep` argument.

150    blocks would cancel each other out in pairs when repeating the optimized circuit).

360        for _ in range(self._circuit_rep):
361            circuit.compose(synthesized_phase_block, inplace=True)
362        circuit.compose(synthesized_cx_block2, inplace=True)
363        if self._reallocate:
364            undo_mapping = [
365                self._qubit_mapping.index(i) for i in range(self.num_qubits)
366            ]
367            initial_layout = [
368                undo_mapping[i] for i in synthesized_cx_block1.metadata["final_layout"]
369            ]
370            final_layout = [
371                undo_mapping[i] for i in synthesized_cx_block2.metadata["final_layout"]
372            ]
373            circuit.metadata = {
374                "initial_layout": initial_layout,
375                "final_layout": final_layout,
376            }

381        Simplifies the phase block according to the commutation and fusion
382        rules for phase gadgets.
383        """
384        new_phase_block = self._phase_block.cloned()

386            new_phase_block >>= self._phase_block

393    def anneal(
394        self,
395        num_iters: int,
396        *,
397        schedule: Union[StandardTempSchedule, TempSchedule] = ("linear", 1.0, 0.1),
398        loggers: AnnealingLoggers = {},
399    ):
400        # pylint: disable = dangerous-default-value
401        # pylint: disable = too-many-locals
402        """
403        Performs a cycle of simulated annealing optimization,
404        using the given number of iterations, temperature schedule,
405        initial/final temperatures.
406
407        The circuit is modified in-place and then returned, as per the
408        [fluent API pattern](https://en.wikipedia.org/wiki/Fluent_interface).
409        """
410        # Validate arguments:
411        if not isinstance(num_iters, int) or num_iters <= 0:
412            raise TypeError(f"Expected a positive integer, found {num_iters}.")
413        schedule = _validate_temp_schedule(schedule)
414        log_start, log_iter, log_end = _validate_loggers(loggers)
415        # Log start:
416        if log_start is not None:
417            log_start(self._cx_count, num_iters)
418        # Pre-sample random numbers to use in iterations:
419        rand = self._rng.uniform(size=num_iters)
420        # Run iterations:
421        for it in range(num_iters):
422            old_output_mapping = [i for i in self._output_mapping]
423            t = schedule(it, num_iters=num_iters)
424            layer_idx, (ctrl, trgt) = self.random_flip_cx()
425            new_cx_count, new_cx_blocks_count = self._compute_cx_count()
426            cx_count_diff = new_cx_count - self._cx_count
427            accept_step = cx_count_diff < 0 or rand[it] < np.exp(
428                -np.log(2) * cx_count_diff / t
429            )
430            if log_iter is not None:
431                log_iter(
432                    it,
433                    self._cx_count,
434                    new_cx_count,
435                    accept_step,
436                    (layer_idx, (ctrl, trgt)),
437                    t,
438                    num_iters,
439                )
440            if accept_step:
441                # Accept changes:
442                self._cx_count = new_cx_count
443                self._cx_blocks_count = new_cx_blocks_count
444            else:
445                # Undo changes:
446                self._flip_cx(layer_idx, ctrl, trgt)
447                if self._reallocate:
448                    self._output_mapping = old_output_mapping
449        # Log end:
450        if log_end is not None:
451            log_end(self._cx_count, num_iters)

453    def random_flip_cx(self) -> Tuple[int, Tuple[int, int]]:
454        """
455        Randomly flips a CX gate in the CX circuit used for the optimization,
456        updating both the CX circuit and the circuit being optimized.
457
458        Returns the layer index and gate (pair of control and target) that were
459        flipped (e.g. in case the flip needs to be subsequently undone).
460        """
461        while True:
462            layer_idx = int(self._rng.integers(len(self._cx_block)))
463            ctrl, trgt = self._cx_block[layer_idx].random_flip_cx(self._rng)
464            if layer_idx < len(self._cx_block) - 1 and self._cx_block[
465                layer_idx + 1
466            ].has_cx(ctrl, trgt):
467                # Try again if CX gate already present in layer above (to avoid redundancy)
468                continue
469            if layer_idx > 0 and self._cx_block[layer_idx - 1].has_cx(ctrl, trgt):
470                # Try again if CX gate already present in layer below (to avoid redundancy)
471                continue
472            self._flip_cx(layer_idx, ctrl, trgt)
473            return layer_idx, (ctrl, trgt)

475    def is_cx_flippable(self, layer_idx: int, ctrl: int, trgt: int) -> bool:
476        """
477        Checks whether the given CX gate can be flipped in the given layer.
478        """
479        if not isinstance(layer_idx, int) or not 0 <= layer_idx < len(self._cx_block):
480            raise TypeError(f"Invalid layer index {layer_idx} for CX circuit.")
481        layer = self._cx_block[layer_idx]
482        return layer.is_cx_flippable(ctrl, trgt)

484    def flip_cx(self, layer_idx: int, ctrl: int, trgt: int):
485        """
486        Performs the actions needed to flip the given CX gate in the given layer
487        of the CX circuit used for the optimization:
488
489        - undoes all gates in layers subsequent to the given layer which are
490          causally following the given gate, starting from the last layer and
491          working backwards towards the gate;
492        - applies the desired gate;
493        - redoes all gate undone, in reverse order (starting from the gate and
494          working forwards towards the last layer).
495        """
496        if not self.is_cx_flippable(layer_idx, ctrl, trgt):
497            raise ValueError(
498                f"Gate {(ctrl, trgt)} cannot be flipped in layer number {layer_idx}."
499            )
500        self._flip_cx(layer_idx, ctrl, trgt)
501
502    def _flip_cx(self, layer_idx: int, ctrl: int, trgt: int) -> None:
503        conj_by: Deque[Tuple[int, int]] = deque([(ctrl, trgt)])
504        qubits_spanned: Set[int] = set([ctrl, trgt])
505        for layer in reversed(self._cx_block[:layer_idx]):
506            new_qubits_spanned: Set[int] = set()
507            incident_gates: Set[Tuple[int, int]] = set()
508            for q in qubits_spanned:
509                incident_gate = layer.incident(q)
510                if incident_gate is not None:
511                    incident_gates.add(incident_gate)
512                    new_qubits_spanned.update({incident_gate[0], incident_gate[1]})
513            for incident_gate in incident_gates:
514                conj_by.appendleft(incident_gate)  # will first undo the gate ...
515                # ... then do all gates already in conj_by ...
516                conj_by.append(incident_gate)  # ... then finally redo the gate
517            qubits_spanned.update(new_qubits_spanned)
518        # Flip the gate in the CX circuit:
519        self._cx_block[layer_idx].flip_cx(ctrl, trgt)
520        # Conjugate the optimized phase gadget circuit by all necessary gates:
521        for cx in conj_by:
522            self._phase_block.conj_by_cx(*cx)

524    def _compute_cx_count(self) -> Tuple[int, int]:
525        # pylint: disable = protected-access
526        self._output_mapping = list(range(self.num_qubits))
527        if self._cx_method == "naive":
528            front_cx_count = self._cx_block.num_gates
529        else:
530            front_cx_count = CXCircuit.from_parity_matrix(
531                self._cx_block.parity_matrix(),
532                self.topology,
533                reallocate=self._reallocate,
534                method=self._cx_method,
535            ).num_gates
536        phase_qiskit, cxs = self._phase_block.to_qiskit(
537            self.topology,
538            simplified=False,
539            method=self._phase_method,
540            cx_synth="naive",
541            reallocate=self._reallocate,
542            return_cx=True,
543        )
544        counts = phase_qiskit.count_ops()
545        if "cx" in counts:
546            phase_cx_count = counts["cx"]
547        else:
548            phase_cx_count = 0
549        cxs >>= self._cx_block
550        if self._cx_method == "naive":
551            new_cxs = cxs
552        else:
553            new_cxs = CXCircuit.from_parity_matrix(
554                cxs.parity_matrix(),
555                self.topology,
556                reallocate=self._reallocate,
557                method=self._cx_method,
558            )
559        back_cx_count = new_cxs.num_gates
560        cx_count = front_cx_count + phase_cx_count + back_cx_count
561        if self._reallocate:
562            self._output_mapping = new_cxs._output_mapping
563        return cx_count, 2 * front_cx_count

Container for a phase circuit to be progressively optimized. The original phase circuit is passed to the constructor, together with a qubit topology and a fixed number of layers constraining the CX circuit to be used for simplification.

To understand the structure of the optimized phase circuit, consider the following code snippet:

    opt_circ = PhaseCircuitCXBlockOptimizer(orig_circ, topology, num_cx_layers)
    # perform optimization using the methods of `opt_circ`
    phase_block = opt_circ.phase_block
    cx_block = opt_circ.cx_block

The optimized circuit is obtained by composing three blocks:

  1. a first block of CX gates, given by cx_block.dag() (the same CX gates of cx_block, but in reverse order);
  2. a central block of phase gadgets, given by phase_block;
  3. a final block of CX gates, given by cx_block.

An optional keyword argument circuit_rep (default: 1) can be passed to the constructor to indicate that the original circuit is to be repeated a certain number of times (default: 1). In the optimized circuit, this is achieved by repeating the phase_block part (at point 2. above) a number of times given by the circuit_rep argument. The first and last CX blocks are left unaltered (because the intermediate CX blocks would cancel each other out in pairs when repeating the optimized circuit).

396        *,

398        loggers: AnnealingLoggers = {},
399    ):

401        # pylint: disable = too-many-locals
402        """
403        Performs a cycle of simulated annealing optimization,
404        using the given number of iterations, temperature schedule,
405        initial/final temperatures.
406
407        The circuit is modified in-place and then returned, as per the
408        [fluent API pattern](https://en.wikipedia.org/wiki/Fluent_interface).
409        """
410        # Validate arguments:
411        if not isinstance(num_iters, int) or num_iters <= 0:
412            raise TypeError(f"Expected a positive integer, found {num_iters}.")
413        schedule = _validate_temp_schedule(schedule)
414        log_start, log_iter, log_end = _validate_loggers(loggers)
415        # Log start:
416        if log_start is not None:
417            log_start(self._cx_count, num_iters)
418        # Pre-sample random numbers to use in iterations:
419        rand = self._rng.uniform(size=num_iters)
420        # Run iterations:
421        for it in range(num_iters):
422            old_output_mapping = [i for i in self._output_mapping]
423            t = schedule(it, num_iters=num_iters)
424            layer_idx, (ctrl, trgt) = self.random_flip_cx()
425            new_cx_count, new_cx_blocks_count = self._compute_cx_count()
426            cx_count_diff = new_cx_count - self._cx_count
427            accept_step = cx_count_diff < 0 or rand[it] < np.exp(
428                -np.log(2) * cx_count_diff / t
429            )
430            if log_iter is not None:
431                log_iter(
432                    it,
433                    self._cx_count,
434                    new_cx_count,
435                    accept_step,
436                    (layer_idx, (ctrl, trgt)),
437                    t,
438                    num_iters,
439                )
440            if accept_step:
441                # Accept changes:
442                self._cx_count = new_cx_count
443                self._cx_blocks_count = new_cx_blocks_count
444            else:
445                # Undo changes:
446                self._flip_cx(layer_idx, ctrl, trgt)
447                if self._reallocate:
448                    self._output_mapping = old_output_mapping
449        # Log end:
450        if log_end is not None:
451            log_end(self._cx_count, num_iters)
